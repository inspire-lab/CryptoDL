48 template <typename Pixel = uint8_t, typename Label = uint8_t>
49 std::vector<std::vector<Pixel>> read_mnist_image_file(const std::string& path) {
50  auto buffer = read_mnist_file(path, 0x803);
51 
52  if (buffer) {
53  auto count = read_header(buffer, 1);
54  auto rows = read_header(buffer, 2);
55  auto columns = read_header(buffer, 3);
56 
57  //Skip the header
58  //Cast to unsigned char is necessary cause signedness of char is
59  //platform-specific
60  auto image_buffer = reinterpret_cast<unsigned char*>(buffer.get() + 16);
61 
62  std::vector<std::vector<Pixel>> images;
63  images.reserve(count);
64 
65  for (size_t i = 0; i < count; ++i) {
66  images.emplace_back(rows * columns);
67 
68  for (size_t j = 0; j < rows * columns; ++j) {
69  auto pixel = *image_buffer++;
70  images[i][j] = static_cast<Pixel>(pixel);
71  }
72  }
73 
74  return images;
75  }
76 
77  return {};
78 }
